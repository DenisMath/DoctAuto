Робота продовжує дослідження 2-адичних групових автоматів за
допомогою представлення 2-адичними ізометріями. Дослідження
ізометрій кільця $Z_2$ є важливим з точки зору теорії
 групових автоматів, оскільки кожен двійковий груповий автомат можна розглядати, як таку
 ізометрію. Важливий клас ізометрій утворюють поліноміальні
 функції, дослідженню яких на ізометричність і присвячена дана стаття.


 Отже, розглянемо многочлени кільця
$Z_2[x]$.
 У цьому розділі буде сформульовано умови, при яких многочлен
 $f(x)=a_0+a_1x+a_2x^2+...+a_nx^n$ буде ізометрією кільця $Z_2$.
 \begin{definition}
Означимо $S_n(x_1,x_2)$, як  $$S_n(x_1,x_2)=\sum_{k=0}^{n-1}
x_1^{n-k-1}\cdot x_2^k $$
 \end{definition}


 \begin{example} $S_1(x_1,x_2)=1$, $S_2(x_1,x_2)=x_1+x_2$, $S_3(x_1,x_2)=x_1^2+x_1\cdot x_2+x_2^2\ $ і т.д.
 \end{example}
 \begin{definition}  Означимо функцію $\mu(x)=\overline{ x}$:

$$
\mu(x)=\begin{cases}
0, x\in 2Z_2\\

1,x\in Z_2^*
\end{cases}
$$
 \end{definition}



 \begin{lemma}\label{lem1}
 $\overline{S_{2k}}(x_1,x_2)=\overline{ x}_1\oplus \overline{ x}_2, k\in \mathbb{N}$
 \end{lemma}
 \begin{proof}
 $S_{2k}$ складається з с парної кількості доданків, кожен з яких буде непарним,  якщо $x_1$ та $x_2$ непарні. Тому  $S_{2k}$ є парним, якщо $x_1$ та $x_2$ непарні. Очевидно, що $S_{2k}$ є парним, якщо $x_1$ та $x_2$ є парними. Крім того $S_{2k}=x_1+S_{2k-1}\cdot x_2=S_{2k-1}\cdot x_1+x_2$, тому $S_{2k}$ є непарним, якщо $x_1$ та $x_2$ - різної парності. Отже таблиця значень для $\overline{S_{2k}}(x_1,x_2)$ співпадає з таблицею істинності для $\overline{ x}_1\oplus \overline{ x}_2$.
 \end{proof}
 \begin{lemma}\label{lem2}
 $\overline{S_{2k+1}}(x_1,x_2)=\overline{ x}_1\cup \overline{ x}_2, k\in \mathbb{N}$
 \end{lemma}

 \begin{proof}
 $S_{2k+1}$ складається з с непарної кількості доданків, кожен з яких буде непарним,
   якщо $x_1$ та $x_2$ непарні. Тому  $S_{2k}$ є непарним,
   якщо $x_1$ та $x_2$ непарні. Очевидно, що $S_{2k}$ є парним, якщо $x_1$ та $x_2$
   є парними. Крім того $S_{2k}=x_1+S_{2k-1}\cdot x_2=S_{2k-1}\cdot x_1+x_2$,
   тому $S_{2k}$ є непарним, якщо $x_1$ та $x_2$ - різної парності. Отже таблиця
   значень для $\overline{S_{2k+1}}(x_1,x_2)$ співпадає з таблицею істинності для
    $\overline{ x}_1\cup \overline{ x}_2$.
 \end{proof}
  \begin{definition}
 $$D_f(x_1,x_2)=\frac{f(x_1)-f(x_2)}{x_1-x_2}$$
 \end{definition}
 \begin{lemma}\label{rivnsum}
 Для многочлена $f(x)=a_0+a_1x+a_2x^2+...+a_nx^n$
  має місце рівність $$D_f(x_1,x_2)=\sum_{k=1}^n a_kS_k(x_1,x_2)$$
 \end{lemma}
 \begin{proof}
Дійсно,
$f(x_1)-f(x_2)=(a_0+a_1x_1+a_2x_1^2+...+a_nx_1^n)-(a_0+a_1x_2+a_2x_2^2+...+a_nx_2^n)=
\sum_{k=1}^n a_k(x_1^k-x_2^k)$.

Очевидно, що для всіх $k \in \mathbb{N}$
$$(x_1^k-x_2^k)/(x_1-x_2)=\sum_{t=0}^{k-1}
x_1^{k-t-1}\cdot x_2^t =S_k(x_1,x_2)$$

Звідси маємо потрібну рівність: $$D_f(x_1,x_2)=\sum_{k=1}^n
a_kS_k(x_1,x_2)$$
 \end{proof}
 \begin{lemma}\label{izomumov1}
 Многочлен $f(x)\in Z_2[x]$ є ізометрією тоді, і тільки тоді,
 коли $$\forall\ x_1,x_2 \in Z_2\  \overline{D_f}(x_1,x_2)=1$$
 \end{lemma}
 \begin{proof}
 Дійсно, оскільки $f$ - ізометрія, то $(f(x_1)-f(x_2))$ та
 $(x_1-x_2)$ мають однакову кількість 0 на
 початку двійкового запису, а отже їх відношення буде непарним
 2-адичним числом.
 \end{proof}
 \begin{definition}
 Для многочлена $f(x)=a_0+a_1x+a_2x^2+...+a_nx^n$ означимо $A_f$ та $B_f$:
 $$ A_f=\mu \left(\sum_{k=1}^{[\frac{n+1}{2}]}a_{2k}\right)$$
 $$ B_f=\mu \left(\sum_{k=2}^{[\frac{n+1}{2}]}a_{2k-1}\right)$$
  \end{definition}
  \begin{theorem}\label{rivnist} Має місце наступна рівність:
 $$\overline{D_f}(x_1,x_2)=\overline{ a}_1\oplus (A_f\cap (\overline{
 x}_1
 \oplus \overline{ x}_2))\oplus (B_f\cap (\overline{
 x}_1\cup \overline{
 x}_2))$$
  \end{theorem}
 \begin{proof}
За лемою \ref{rivnsum} маємо:
 $$\overline{D_f}(x_1,x_2)=\overline{\sum_{k=1}^n a_kS_k (x_1,x_2)}$$
Далі, оскільки
$\overline{a+b}=\overline{\phantom{b}a}\oplus \overline{\phantom{b}b}$, то:
 $$\overline{\sum_{k=1}^n a_kS_k (x_1,x_2)}=\oplus \kern -.5em \sum_{k=1}^n\overline{ a_kS_k (x_1,x_2)}$$
 оскільки
$\overline{a*b}=\overline{\phantom{b}a}*\overline{\phantom{b}b}$, то:
 $$\overline{\sum_{k=1}^n a_kS_k (x_1,x_2)}=\oplus \kern -.45em \sum_{k=1}^n\overline{a}_k\overline{S_k (x_1,x_2)} \ \ \ (*)$$
Скориставшись лемами \ref{lem1} та \ref{lem2} отримуємо з $(*)$ потрібну рівність.
%$$\oplus \kern -.5em \sum_{k=1}^n\overline{a}_k\overline{S_k (x_1,x_2)}=\oplus \kern -.65em \sum_{k=1}^{[\frac{n+1}{2}]}a_{2k}\bigoplus \oplus \kern -.65em \sum_{k=2}^{[\frac{n+1}{2}]}a_{2k-1}$$

 \end{proof}

  \begin{corollary}\label{naslidok1} Згідно з лемою \ref{izomumov1}
   та теоремою \ref{rivnist}
  многочлен $f(x)=a_0+a_1x+a_2x^2+...+a_nx^n$ є ізометрією тоді, і тільки тоді,
  коли формула
  $\overline{ a}_1\oplus (A_f\cap (\overline{x}_1\oplus x_2))\oplus (B_f\cap (\overline{x}_1\cup \overline{x}_2))$
  є тотожньо-істиною.
   \end{corollary}
   \begin{theorem}\label{istformul}
    $\overline{ a}_1\oplus (A_f\cap (\overline{x}_1\oplus \overline{x}_2))\oplus (B_f\cap (\overline{x}_1\cup \overline{x}_2))$
    є тотожньо-істиною тоді, і тільки тоді, коли $\overline{ a}_1=1,\ A_f=0,\ B_f=0$
  \end{theorem}
  \begin{proof} Побудуємо табличку вигляду формули
   $\overline{ a}_1\oplus (A_f\cap (\overline{x}_1\oplus \overline{x}_2))\oplus (B_f\cap (\overline{x}_1\cup \overline{x}_2))$
   для різних значень $\overline{ a}_1,\ A_f,\ B_f$.

  \begin{center}



  \begin{tabular}{|c|c|c|c|}
\hline
$ \overline{ a}_1$ & $A_f$ & $B_f$ & Результат \\
\hline
\hline
$ 0$ & $0$ & $0$ & 0\\
\hline

$ 0$ & $0$ & $1$ & $\overline{x}_1\cup \overline{x}_2$\\
\hline

$ 0$ & $1$ & $0$ & $\overline{x}_1\oplus \overline{x}_2$\\
\hline

$ 0$ & $1$ & $1$ & $(\overline{x}_1\oplus \overline{x}_2)\oplus (\overline{x}_1\cup \overline{x}_2)$\\
\hline

$ 1$ & $0$ & $0$ & 1\\
\hline

$ 1$ & $0$ & $1$ & $\neg(\overline{x}_1\cup \overline{x}_2)$\\
\hline

$ 1$ & $1$ & $0$ & $\neg(\overline{x}_1\oplus \overline{x}_2)$\\
\hline

$ 1$ & $1$ & $1$ & $\neg((\overline{x}_1\oplus \overline{x}_2)\oplus (\overline{x}_1\cup \overline{x}_2))$\\
\hline


\end{tabular}

\end{center}

Таблиці істинності формул $\overline{x}_1\cup \overline{x}_2$,
$\overline{x}_1\oplus \overline{x}_2$ та $(\overline{x}_1\oplus
\overline{x}_2)\oplus (\overline{x}_1\cup \overline{x}_2)$ мають
наступний вигляд:


\begin{center}
\begin{tabular}{|c|c|c|c|c|}

\hline
$\overline{x}_1$  &  $\overline{x}_2$ & $\overline{x}_1\oplus \overline{x}_2$  & $\overline{x}_1\cup \overline{x}_2$&  $(\overline{x}_1\oplus \overline{x}_2)\oplus $\\
\ & \ & \ & \ & $(\overline{x}_1\cup \overline{x}_2)$\\
 \hline \hline
0 & 0 & 0  & 0&  0\\
\hline
0 & 1 & 1  & 1&  0\\
\hline
1 & 0 & 1  & 1&  0\\
\hline
1 & 1 & 0  & 1&  1\\

Отже формула є тотожньо-хибною для $$ \overline{ a}_1=0,\ A_f=0,\
B_f=0$$ тотожньо-істиною для $$ \overline{ a}_1=1,\ A_f=0,\
B_f=0$$ і виконливою у всіх інших випадках.
  \end{proof}
   \begin{theorem}\label{naslidok2} Згідно з наслідком \ref{naslidok1} та теоремою \ref{istformul} многочлен $f(x)=a_0+a_1x+a_2x^2+...+a_nx^n$ є ізометрією тоді, і тільки тоді, коли коефіціент $a_1$ є непарним 2-адичним числом, сума коефіціентів з парними номерами більше 0 є парним 2-адичним числом та сума коефіціентів з непарними номерами більше 3 є парним 2-адичним числом
   \end{theorem}
   \begin{example}
   Згідно з теоремою \ref{naslidok2} наступні многочлени  є
   ізометріями: $$f(x)=5x+1$$ $$f(x)=x^4+x^2+x$$ $$f(x)=x^5+2x^4+x^3+3x $$
   а многочлени $$f(x)=4x+1$$ $$f(x)=x^3+x^2+x$$ $$f(x)=x^5+x^4+x^3+3x $$ не є
   ізометріями кільця $Z_2$.
   \end{example}
